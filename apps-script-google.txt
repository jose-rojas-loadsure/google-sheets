/** CONFIGURACION INICIAL PARA TRAER LA BASE DE DATOS */

function ordenarAutomaticamente() {
  const libro=SpreadsheetApp.getActiveSpreadsheet();
  const hoja=libro.getSheetByName("Datos")
  const arregloOriginal=hoja.getRange(2,1,hoja.getLastRow()-1,hoja.getLastColumn()).getValues()
  Logger.log(arregloOriginal)
  arregloOriginal.sort(funcionOrdenarStatusAscendente)
  Logger.log(arregloOriginal)
  hoja.getRange(2,1,hoja.getLastRow()-1,hoja.getLastColumn()).setValues(arregloOriginal)
}

/** COLUMNA STATUS */

/** Funcionalidad para ordenar de menor a mayor de forma alfabetica de la [A-Z] tomando como referencia la columna Status */
var funcionOrdenarStatusAscendente=function(a,b){
  if(a[5]>b[5]){return 1}
  else if(b[5]>a[5]){return -1}
  return 0
}


/** Funcionalidad para ordenar de mayor a menor de forma alfabetica de la [A-Z] tomando como referencia la columna Status */
var funcionOrdenarStatusDescendente=function(a,b){
  if(a[5]>b[5]){return -1}
  else if(b[5]>a[5]){return 1}
  return 0
}
/**_______________________________________________________________________________________________________________________ */

/** COLUMNA FECHA SOLICITUD */

var funcionOrdenarFechaSolicitudAscendente=function(a,b){

  return 0

var funcionOrdenarFechaSolicitudDescendente=function(a,b){

  else if(b[3]>a[3]){return 1}
  return 0
}
/**_______________________________________________________________________________________________________________________ */

/** COLUMNA FECHA CULMINACION */

var funcionOrdenarFechaCulminacionAscendente=function(a,b){
  if(a[4]>b[4]){return 1}
  else if(b[4]>a[4]){return -1}
  return 0
}

var funcionOrdenarFechaCulminacionDescendente=function(a,b){
  if(a[4]>b[4]){return -1}
  else if(b[4]>a[4]){return 1}
  return 0
}
